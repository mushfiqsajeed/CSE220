private int hashFunction( Integer key ){
        int len = ht.length;
        
        int k = (key+3) % len;

        return k;
    }

public void remove( Integer key ){

        int indx = hashFunction(key);

        if (indx<0 || indx > (ht.length-1))
        return;

        PairNode head = ht[indx];
        PairNode current = head;

        while (current != null)
        {
            if (current.key==key) //Head remove korar condition
            {
                ht[indx] = ht[indx].next;
                break;
            }

            if (current.next != null && current.next.key == key)
            {
                PairNode nxt = current.next;
                current.next = nxt.next;
                break;
            }

            else
            current = current.next;
        }
    }

public static String removeConsecDups(String word){
	// You MUST create a LinkedListQueue Object to solve this task
        LinkedListQueue q = new LinkedListQueue();
        int len = word.length();
        String new_word = "";

        for(int i=0; i<len; i++)
        {
            char temp = word.charAt(i);

            if (q.isEmpty())
            {
                q.enqueue(temp);
                new_word += temp;
            }

            else
            {
                char x = (char) q.dequeue();

                if (temp != x)
                {
                    q.enqueue(temp);
                    new_word += temp;
                }
                else
                q.enqueue(x);
            }
        }

	return new_word; //remove this line once you're ready
    }
